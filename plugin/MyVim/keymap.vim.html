 395     let lin_right = line("'>")
 396     let sav_pos   = getpos('.')
 397     let left  = [lin_left,col_left]

 399     if a:mode is 'visual'
 400        call cursor(left)
 401     elseif a:mode is 'normal'
 402        call setpos('.',sav_pos)
 403     endif
 404     normal i
 405     if a:mode is 'visual'
 406        call cursor(right)
 407     elseif a:mode is 'normal'
 408        call setpos('.',sav_pos)
 409     endif
 410     normal l
 411     normal a
